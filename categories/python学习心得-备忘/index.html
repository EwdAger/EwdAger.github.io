每一趟遍历把最小的依次放在最前面，时间复杂度O(n²)

def select_sort(n):
    for i in range(len(n):
        min = i
        for j in range(i, len(n)):
            if n[min] > n[j]:
                min = j
        if min != i:
            n[min], n[i] = n[i], n[min]
    return n
